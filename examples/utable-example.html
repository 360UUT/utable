	  ...
		   FooterOptions: {
            Rows: [
                {
                    RowType: "合计",
                    RowTypeNameIndex: 2,
                    Cells: [
                        {
                            CellIndex: 8,
                            // data 为请求回来的数据, cell为当前单元格Dom
                            Func: function (data, cell) {
                             //   console.log(data);
                                var total=0;
                                for(var i=0;i<data.Result.Items.length;i++){
                                    total+=data.Result.Items[i].money;
                                }

                                return total;
                            }
                        }
                    ]
                },
                {
                    RowType: "平均数",
                    RowTypeNameIndex: 2,
                    Cells: [
                        {
                            CellIndex: 8,
                            // data 为请求回来的数据, cell为当前单元格Dom
                            Func: function (data, cell) {
                                var total=0;
                                for(var i=0;i<data.Result.Items.length;i++){
                                    total+=data.Result.Items[i].money;
                                }

                                return total/data.Result.Items.length;
                            }
                        }
                    ]
                }
            ]
        },
		...
	});
